Creating the Model

In[21]:
# The transformed price feature is extracted form the input variable set X, and placed as the target variable y
X = data.drop(columns=['transformed_price'])
y = data['transformed_price']
In[22]:
# The processed dataset is split into training and test set, with test set being 1/5th of its parent dataset
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=0)
In[23]:
# Use the standardscaler to standardise the x train and x test sets according to their mean and standard deviation
scaler = StandardScaler()
X_train_scaled = scaler.fit_transform(X_train)
X_test_scaled = scaler.transform(X_test)
In[24]:
# Apply a random forest on the training dataset 
rf_algorithm2 = RandomForestRegressor(random_state=42)
rf_algorithm2.fit(X_train_scaled, y_train)
# The trained regresser model is then used to predict output values for the given test set
predictions = rf_algorithm2.predict(X_test_scaled)
In[25]:
# Determining the MSE (mean squred error) between actual and predicted scores, an MSE as low as 7.38e-05 indicates strong model accuracy
mse_rf = mean_squared_error(y_test, predictions)
# Similarly the R2 score, obtained to be around 0.87 highlights the goodness of the model fit to the test data values,
# given it is approximate to the max R2 score of 1.
r2_rf = r2_score(y_test, predictions)

print("Random Forest Results:")
print(f"Mean Squared Error (MSE): {mse_rf}")
print(f"R Score: {r2_rf}")
